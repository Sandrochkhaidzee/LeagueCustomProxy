# v0.3.0 Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use `superpowers:executing-plans` to implement this plan task-by-task.

**Goal:** Ship v0.3.0 — server-side tiered proximity (team voice always full, cross-team default-600u-toggle-to-1200u), plus #11 debug thumbnail dynamic resize, #1 PTT low-level keyboard hook with rebindable key, CV tracking code tweaks driven by issue #7 logs, and an optional CV model retrain.

**Architecture:** Server-side filter for cross-team volumes (team filter cannot be bypassed by a modified client). New wire fields: `team` on WSS `join`, `hearCrossTeam` piggybacked on every `coords` message. Server reads both from room state, not request body. Client-side: Settings toggle persists in localStorage, sent on join + every coords tick. Back-compat: server falls back to legacy v0.2 behavior when `team` is missing. Independent workstreams — any phase can slip to v0.3.1 without blocking the rest.

**Tech Stack:** TypeScript (client + server), Rust + Tauri 2 (desktop app), Jest (client tests), Vitest (server tests), Win32 API (`SetWindowsHookExW`, `SendInput`).

**Design doc:** `docs/plans/2026-06-02-v0.3.0-design.md` (read this before starting).

```typescript
export const DEFAULT_CROSS_TEAM_RANGE = 600;

export interface TieredRoomClient {
  name: string;
  team?: 'ORDER' | 'CHAOS';
  hearCrossTeam?: boolean;
  position?: { x: number; y: number; updatedMs: number };
}

/**
 * v0.3 tiered proximity. Server-authoritative team filter + cross-team
 * range cap. The full 1200u falloff curve is used in both modes — only
 * the cutoff changes, so distance X sounds the same regardless of toggle.
 */
export function computeTieredVolumes(
  body: VolumeRequestV2,
  getRoomClients: (roomId: string) => TieredRoomClient[],
): VolumeResponse {
  // Validate (mirrors computeVolumesFromRoom)
  if (!body.myPosition || typeof body.myPosition.x !== 'number' || typeof body.myPosition.y !== 'number' ||
      !isFinite(body.myPosition.x) || !isFinite(body.myPosition.y)) {
    throw new Error('Invalid position');
  }
  if (typeof body.roomId !== 'string' || !body.roomId) throw new Error('Invalid roomId');
  if (typeof body.name !== 'string' || !body.name) throw new Error('Invalid name');

  const clients = getRoomClients(body.roomId);
  const me = clients.find(c => c.name === body.name);
  if (!me) return { myBlob: '', peerVolumes: {} };

  // Legacy v0.2 fallback: missing team → ignore team filter, use 1200u
  const legacy = me.team === undefined;
  const range = legacy
    ? MAX_HEARING_RANGE
    : (me.hearCrossTeam ? MAX_HEARING_RANGE : DEFAULT_CROSS_TEAM_RANGE);

  const cutoff = Date.now() - STALE_POSITION_MS;
  const peerVolumes: Record<string, number> = {};

  for (const peer of clients) {
    if (peer.name === me.name) continue;

    // Allies always full
    if (!legacy && peer.team === me.team) {
      peerVolumes[peer.name] = 1.0;
      continue;
    }

    // Cross-team (or legacy "all team-blind") — needs valid fresh position
    if (!peer.position) continue;
    if (peer.position.updatedMs < cutoff) continue;

    const dx = body.myPosition.x - peer.position.x;
    const dy = body.myPosition.y - peer.position.y;
    const dist = Math.sqrt(dx * dx + dy * dy);
    if (dist >= range) continue;

    // Hard-cut at `range`, but use the full 1200u curve so loudness is
    // consistent across toggle states.
    peerVolumes[peer.name] = calculateVolume(dist);
  }

  return { myBlob: '', peerVolumes };
}
```

Also export `MAX_HEARING_RANGE` if not already exported (test imports it via `calculateVolume`).

```typescript
 * Map between browser `KeyboardEvent.code` strings and Win32 virtual-key
 * codes for the PTT/toggle-mute rebind UI.
 *
 * Only covers the keys we want to allow as bindings — leaves out hostile
 * choices (Esc, Tab, modifier-as-binding) and rare keys.

const KEY_TO_VK: Record<string, number> = {
  // Letters A-Z → 0x41-0x5A
  ...Object.fromEntries(
    Array.from({ length: 26 }, (_, i) => [`Key${String.fromCharCode(65 + i)}`, 0x41 + i]),
  ),
  // Digits 0-9 → 0x30-0x39
  ...Object.fromEntries(
    Array.from({ length: 10 }, (_, i) => [`Digit${i}`, 0x30 + i]),
  ),
  // F1-F12 → 0x70-0x7B
  ...Object.fromEntries(
    Array.from({ length: 12 }, (_, i) => [`F${i + 1}`, 0x70 + i]),
  ),
  CapsLock: 0x14,
  Insert: 0x2D,
  Delete: 0x2E,
  Home: 0x24,
  End: 0x23,
  PageUp: 0x21,
  PageDown: 0x22,
  Space: 0x20,
  Backquote: 0xC0,    // `
  Backslash: 0xDC,    // \
};

  Object.entries(KEY_TO_VK).map(([code, vk]) => {
    let label: string;
    if (code.startsWith('Key')) label = code.slice(3);
    else if (code.startsWith('Digit')) label = code.slice(5);
    else if (/^F\d+$/.test(code)) label = code;
    else if (code === 'CapsLock') label = 'Caps Lock';
    else if (code === 'PageUp') label = 'Page Up';
    else if (code === 'PageDown') label = 'Page Down';
    else if (code === 'Backquote') label = '`';
    else if (code === 'Backslash') label = '\\';
    else label = code;
    return [vk, label];
  }),
);

export function browserKeyToWin32Vk(code: string): number | null {
  return KEY_TO_VK[code] ?? null;
}

export function humanizeVk(vk: number): string {
  return VK_TO_HUMAN[vk] ?? 'Unknown';
}
```

**Step 2: Create global_keys.rs.** This is a large file — see the design doc Section 4 for the full architecture. Key requirements:
- Static atomics for PTT_VK + TOGGLE_VK
- Hook installed from a dedicated `std::thread` with a Win32 message pump
- Hook callback posts to `tokio::sync::mpsc::UnboundedSender<KeyEvent>` and returns immediately
- Worker tokio task drains the channel and calls `app_handle.emit("ptt", ...)` / `emit("toggle_self_mute", ...)`
- Caps Lock LED workaround: on Caps Lock event, send synthetic `SendInput(VK_CAPITAL, down+up)` via `windows::Win32::UI::Input::KeyboardAndMouse::SendInput`

Skeleton:
```rust
use std::sync::atomic::{AtomicU32, Ordering};
use tauri::{AppHandle, Emitter};
use tokio::sync::mpsc;
use windows::Win32::UI::WindowsAndMessaging::{
    SetWindowsHookExW, UnhookWindowsHookEx, CallNextHookEx, GetMessageW,
    WH_KEYBOARD_LL, HHOOK, KBDLLHOOKSTRUCT, MSG,
    WM_KEYDOWN, WM_KEYUP, WM_SYSKEYDOWN, WM_SYSKEYUP,
};
use windows::Win32::UI::Input::KeyboardAndMouse::{
    SendInput, INPUT, INPUT_KEYBOARD, KEYBDINPUT, KEYEVENTF_KEYUP, VIRTUAL_KEY,
};

const VK_CAPITAL: u32 = 0x14;

static PTT_VK:    AtomicU32 = AtomicU32::new(VK_CAPITAL);  // default: Caps Lock
static TOGGLE_VK: AtomicU32 = AtomicU32::new(0);            // default: unbound

// SAFETY: HHOOK is `*mut c_void` under the hood and is NOT Send by default.
// Hook ownership stays on the dedicated hook thread, so we never share this
// across threads — but Rust still requires Sync for `static`. Wrap the pointer
// in an opaque struct + impl Sync ourselves. The atomicity of the install/uninstall
// lifecycle is guaranteed by the once-only setup in setup_hook below.
struct HookSlot(std::cell::UnsafeCell<HHOOK>);
unsafe impl Sync for HookSlot {}
static HOOK: HookSlot = HookSlot(std::cell::UnsafeCell::new(HHOOK(std::ptr::null_mut())));

// mpsc sender — set once on init, used from the hook callback to post events
// back to the async runtime.
static EVENT_TX: std::sync::OnceLock<mpsc::UnboundedSender<KeyEvent>> = std::sync::OnceLock::new();

#[derive(Debug, Clone)]
enum KeyEvent { PttDown, PttUp, ToggleMute }

unsafe extern "system" fn hook_proc(code: i32, wparam: windows::Win32::Foundation::WPARAM, lparam: windows::Win32::Foundation::LPARAM) -> windows::Win32::Foundation::LRESULT {
    if code >= 0 {
        let kb = &*(lparam.0 as *const KBDLLHOOKSTRUCT);
        let msg = wparam.0 as u32;
        let is_down = matches!(msg, WM_KEYDOWN | WM_SYSKEYDOWN);
        let is_up   = matches!(msg, WM_KEYUP   | WM_SYSKEYUP);
        let ptt_vk    = PTT_VK.load(Ordering::Relaxed);
        let toggle_vk = TOGGLE_VK.load(Ordering::Relaxed);

        if let Some(tx) = EVENT_TX.get() {
            if kb.vkCode == ptt_vk {
                if is_down { let _ = tx.send(KeyEvent::PttDown); }
                else if is_up { let _ = tx.send(KeyEvent::PttUp); }
                // Caps Lock LED workaround
                if ptt_vk == VK_CAPITAL && (is_down || is_up) {
                    flip_caps_lock_back();
                }
            } else if kb.vkCode == toggle_vk && is_down {
                let _ = tx.send(KeyEvent::ToggleMute);
            }
        }
    }
    CallNextHookEx(*HOOK.0.get(), code, wparam, lparam)
}

unsafe fn flip_caps_lock_back() {
    let mut inputs = [
        INPUT {
            r#type: INPUT_KEYBOARD,
            Anonymous: windows::Win32::UI::Input::KeyboardAndMouse::INPUT_0 {
                ki: KEYBDINPUT {
                    wVk: VIRTUAL_KEY(VK_CAPITAL as u16),
                    wScan: 0, dwFlags: Default::default(), time: 0, dwExtraInfo: 0,
                },
            },
        },
        INPUT {
            r#type: INPUT_KEYBOARD,
            Anonymous: windows::Win32::UI::Input::KeyboardAndMouse::INPUT_0 {
                ki: KEYBDINPUT {
                    wVk: VIRTUAL_KEY(VK_CAPITAL as u16),
                    wScan: 0, dwFlags: KEYEVENTF_KEYUP, time: 0, dwExtraInfo: 0,
                },
            },
        },
    ];
    SendInput(&inputs, std::mem::size_of::<INPUT>() as i32);
}

pub fn setup_hook(app: AppHandle) {
    let (tx, mut rx) = mpsc::unbounded_channel::<KeyEvent>();
    let _ = EVENT_TX.set(tx);

    // Worker task — emits to Tauri event bus
    let app_clone = app.clone();
    tauri::async_runtime::spawn(async move {
        while let Some(ev) = rx.recv().await {
            match ev {
                KeyEvent::PttDown   => { let _ = app_clone.emit("ptt", serde_json::json!({ "down": true })); }
                KeyEvent::PttUp     => { let _ = app_clone.emit("ptt", serde_json::json!({ "down": false })); }
                KeyEvent::ToggleMute => { let _ = app_clone.emit("toggle_self_mute", serde_json::json!({})); }
            }
        }
    });

    // Hook thread — owns the hook + runs a message pump
    std::thread::spawn(|| unsafe {
        let h = SetWindowsHookExW(WH_KEYBOARD_LL, Some(hook_proc), None, 0)
            .expect("Failed to install WH_KEYBOARD_LL hook");
        *HOOK.0.get() = h;
        let mut msg = MSG::default();
        while GetMessageW(&mut msg, None, 0, 0).as_bool() {
            // No-op — the hook runs on its own; we just need the thread to have
            // a message pump for low-level hooks to fire.
        }
        let _ = UnhookWindowsHookEx(h);
    });
}

#[tauri::command]
pub fn set_ptt_key(vk: u32)    { PTT_VK.store(vk, Ordering::Relaxed); }
#[tauri::command]
pub fn set_toggle_key(vk: u32) { TOGGLE_VK.store(vk, Ordering::Relaxed); }
```

```typescript
import { browserKeyToWin32Vk, humanizeVk } from '../core/keymap';

const FORBIDDEN_KEYS = new Set(['Escape', 'Tab', 'KeyQ', 'KeyW', 'KeyE', 'KeyR']);  // LoL bindings

function setupBindButton(buttonId: string, storageKey: string, backgroundCmd: string) {
  const btn = document.getElementById(buttonId) as HTMLButtonElement;
  const stored = localStorage.getItem(storageKey);
  if (stored) btn.textContent = humanizeVk(parseInt(stored, 10));

  btn.addEventListener('click', () => {
    const original = btn.textContent;
    btn.textContent = 'Press a key…';
    btn.disabled = true;
    const onKey = (e: KeyboardEvent) => {
      e.preventDefault();
      if (FORBIDDEN_KEYS.has(e.code)) {
        btn.textContent = '(common LoL binding — pick another)';
        setTimeout(() => { btn.textContent = original; btn.disabled = false; }, 1500);
        window.removeEventListener('keydown', onKey, true);
        return;
      }
      const vk = browserKeyToWin32Vk(e.code);
      if (vk === null) {
        btn.textContent = '(key not supported)';
        setTimeout(() => { btn.textContent = original; btn.disabled = false; }, 1500);
        window.removeEventListener('keydown', onKey, true);
        return;
      }
      localStorage.setItem(storageKey, String(vk));
      sendToBackground(backgroundCmd, { vk });
      btn.textContent = humanizeVk(vk);
      btn.disabled = false;
      window.removeEventListener('keydown', onKey, true);
    };
    window.addEventListener('keydown', onKey, true);
  });
}

setupBindButton('btn-bind-ptt', 'lolproxchat.pttVk', 'setPttKey');
setupBindButton('btn-bind-toggle', 'lolproxchat.toggleVk', 'setToggleKey');
```
